496template<
497 class T,
498 bool (T::*Send)(const uint8_t*, size_t),
499 bool (T::*Recv)(uint8_t*, size_t, Timeout, size_t*, Timestamp*),
500 bool (T::*Update)(Timeout)
501>
502void DeviceInterface::setCallbacks(T* object)
503{
504 auto send = [](C::mip_interface* device, const uint8_t* data, size_t size)
505 {
506 return (static_cast<T*>(mip_interface_user_pointer(device))->*Send)(data, size);
507 };
508 auto recv = [](C::mip_interface* device, uint8_t* buffer, size_t max_length, C::timeout_type wait_time, size_t* length_out, C::timestamp_type* timestamp_out)
509 {
510 return (static_cast<T*>(mip_interface_user_pointer(device))->*Recv)(buffer, max_length, wait_time, length_out, timestamp_out);
511 };
512 auto update = [](C::mip_interface* device, C::timeout_type wait_time)
513 {
514 return (static_cast<T*>(mip_interface_user_pointer(device))->*Update)(wait_time);
515 };
516
517 C::mip_interface_set_user_pointer(this, object);
518 C::mip_interface_set_send_function(this, Send != nullptr ? send : nullptr);
519 C::mip_interface_set_recv_function(this, Recv != nullptr ? recv : nullptr);
520
521 if( Update != nullptr )
522 C::mip_interface_set_update_function(this, update);
523 else if( Recv != nullptr )
524 C::mip_interface_set_update_function(this, &C::mip_interface_default_update);
525 else
526 C::mip_interface_set_update_function(this, nullptr);
527}
