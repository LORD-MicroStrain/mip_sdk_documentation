499template<
500 class T,
501 bool (T::*Send)(const uint8_t*, size_t),
502 bool (T::*Recv)(uint8_t*, size_t, Timeout, size_t*, Timestamp*),
503 bool (T::*Update)(Timeout)
504>
505void DeviceInterface::setCallbacks(T* object)
506{
507 auto send = [](C::mip_interface* device, const uint8_t* data, size_t size)
508 {
509 return (static_cast<T*>(mip_interface_user_pointer(device))->*Send)(data, size);
510 };
511 auto recv = [](C::mip_interface* device, uint8_t* buffer, size_t max_length, C::mip_timeout wait_time, size_t* length_out, C::mip_timestamp* timestamp_out)
512 {
513 return (static_cast<T*>(mip_interface_user_pointer(device))->*Recv)(buffer, max_length, wait_time, length_out, timestamp_out);
514 };
515 auto update = [](C::mip_interface* device, C::mip_timeout wait_time)
516 {
517 return (static_cast<T*>(mip_interface_user_pointer(device))->*Update)(wait_time);
518 };
519
520 C::mip_interface_set_user_pointer(this, object);
521 C::mip_interface_set_send_function(this, Send != nullptr ? send : nullptr);
522 C::mip_interface_set_recv_function(this, Recv != nullptr ? recv : nullptr);
523
524 if( Update != nullptr )
525 C::mip_interface_set_update_function(this, update);
526 else if( Recv != nullptr )
527 C::mip_interface_set_update_function(this, &C::mip_interface_default_update);
528 else
529 C::mip_interface_set_update_function(this, nullptr);
530}
